\section{Definitions}

\subsection{EBIT TTM}

For quarter $q$:
\[
    \text{EBIT\_TTM}_q = \sum_{i=0}^{3} \text{EBIT}_{q-i}
\]
where $\text{EBIT}_q$ is the quarterly Operating Income.

\subsection{Enterprise Value}

For trading day $t$, let $q(t)$ denote the most recent fiscal quarter with reported financials at or before day $t$.
\[
    \text{EV}_t = \text{Market\_Cap}_t + \text{Total\_Debt}_{q(t)} - \text{Cash\_and\_Equivalents}_{q(t)}
\]
where:
\begin{itemize}
    \item $\text{Market\_Cap}_t = \text{Price}_t \times \text{Shares\_Outstanding}_{q(t)}$
    \item $\text{Total\_Debt}_{q(t)}$ is the total debt at quarter $q(t)$
    \item $\text{Cash\_and\_Equivalents}_{q(t)}$ is the cash and short-term investments at quarter $q(t)$
\end{itemize}

\subsection{EV/EBIT}

For each trading day $t$:
\[
    \text{EV\_EBIT}_t = \frac{\text{EV}_t}{\text{EBIT\_TTM}_{q(t)}}
\]

\section{Signals}

\subsection{Positivity}

\paragraph{Signal:}
\[
    \text{EBIT\_TTM}_q > 0
\]

\subsection{Growth}

For quarter $q$:
\[
    \text{EBIT\_Growth}_q = \frac{\text{EBIT\_TTM}_q}{\text{EBIT\_TTM}_{q-4}} - 1
\]
where $\text{EBIT\_Growth}_q$ is the year-over-year growth rate of EBIT TTM.

\paragraph{Signal:}
\[
    \text{EBIT\_Growth}_q > 0
\]

\subsection{EV/EBIT Cycle}

Let $\{\text{EV\_EBIT}_t\}_{t=1}^{n}$ be the daily EV/EBIT values over $n$ trading days.

\paragraph{25th Percentile (Q1):}
\[
    Q_1 = \text{25th percentile of } \{\text{EV\_EBIT}_t\}_{t=1}^{n}
\]

\paragraph{Signal:}
\[
    \text{EV\_EBIT} < Q_1
\]

\subsubsection{5Y-Cycle}
$n \approx 1250$ (5 years $\times$ $\sim$250 trading days/year)

\subsubsection{1Y-Cycle}
$n \approx 250$ (1 year $\times$ $\sim$250 trading days/year)
