\section{P/E (NTM)}


\subsection{EPS NTM}

For trading day $t$, $\text{EPS\_NTM}_t$ is the split-adjusted Next Twelve Months EPS estimate at day $t$.

\paragraph{Method 1: Quarterly Estimates Interpolation}~
\newline
Let $q$ denote the current fiscal quarter containing $t$.
\[
    \mathrm{EPS\_NTM}_t = w_t \cdot \mathrm{EPS}^{(t)}_q + \sum_{i=1}^{3} \mathrm{EPS}^{(t)}_{q+i} + (1 - w_t) \cdot \mathrm{EPS}^{(t)}_{q+4}
\]
where:
\begin{itemize}
    \item $w_t \in (0, 1]$ is the fraction of quarter $q$ remaining from day $t$
    \item $\mathrm{EPS}^{(t)}_{q+i}$ is the PIT (Point-In-Time) consensus EPS estimate for quarter $q+i$ at day $t$
\end{itemize}

\paragraph{Method 2: Fiscal Year Estimates Interpolation}

\[
    \mathrm{EPS\_NTM}_t = \frac{d_t}{365} \cdot \mathrm{EPS}^{(t)}_{\mathrm{FY1}} + \frac{365 - d_t}{365} \cdot \mathrm{EPS}^{(t)}_{\mathrm{FY2}}
\]
where:
\begin{itemize}
    \item $d_t$ is the number of days remaining from day $t$ to the end of fiscal year $\mathrm{FY1}$
    \item $\mathrm{EPS}^{(t)}_{\mathrm{FY1}}$ is the PIT (Point-In-Time) consensus EPS estimate for the current fiscal year at day $t$
    \item $\mathrm{EPS}^{(t)}_{\mathrm{FY2}}$ is the PIT (Point-In-Time) consensus EPS estimate for the next fiscal year at day $t$
\end{itemize}

\subsection{P/E NTM}

For each trading day $t$:
\[
    \text{PE\_NTM}_t = \frac{\text{Price}_t}{\text{EPS\_NTM}_t}
\]
where $\text{Price}_t$ is the split-adjusted stock price on trading day $t$.
\newline
Note: Both $\text{Price}_t$ and $\text{EPS\_NTM}_t$ must be adjusted to the same split basis.

\subsection{P/E NTM Cycle Signal}

Let $\{\text{PE\_NTM}_t\}_{t=1}^{n}$ be the daily P/E NTM values over $n$ trading days.

\paragraph{25th Percentile (Q1):}
\[
    Q_1 = \text{25th percentile of } \{\text{PE\_NTM}_t\}_{t=1}^{n}
\]

\paragraph{Signal:}
\[
    \text{PE\_NTM} < Q_1
\]

\subsubsection{5Y-Cycle}
$n \approx 1250$ (5 years $\times$ $\sim$250 trading days/year)

\subsubsection{1Y-Cycle}
$n \approx 250$ (1 year $\times$ $\sim$250 trading days/year)

\section{NNI CAGR (1Y TTM)}

\subsection{NI TTM}

For quarter $q$:
\[
    \text{NI\_TTM}_q = \sum_{i=0}^{3} \text{NI}_{q-i}
\]
where $\text{NI\_TTM}_q$ is the Trailing Twelve Months Net Income at quarter $q$.

\subsection{NNI TTM}
\[
    \text{NNI\_TTM}_q = \frac{\text{NI\_TTM}_q}{\text{Shares\_Outstanding}_q}
\]
where $\text{NNI\_TTM}_q$ is the Normalized Net Income (TTM) at quarter $q$.

\subsection{NNI CAGR}
\[
    \text{NNI\_CAGR} = \frac{\text{NNI\_TTM}_q}{\text{NNI\_TTM}_{q-4}} - 1
\]
where $\text{NNI\_CAGR}$ is the Compound Annual Growth Rate of Normalized Net Income.

\paragraph{Signal:}
\[
    \text{NNI\_CAGR} > 0
\]

\subsection{EPS CAGR NTM}
\[
    \text{EPS\_CAGR\_NTM} \approx \frac{\text{EPS\_NTM}_t}{\text{EPS\_TTM}_t} - 1
\]
where $\text{EPS\_CAGR\_NTM}$ is the Compound Annual Growth Rate of EPS NTM.

\paragraph{Signal:}
\[
    \text{EPS\_CAGR\_NTM} > 0
\]

\section{NNI Margin (LTM)}

\subsection{Revenue LTM}
\[
    \text{Revenue\_LTM} = \sum_{i=0}^{3} \text{Revenue}_{q-i}
\]
where $\text{Revenue\_LTM}$ is the Last Twelve Months Revenue.

\subsection{Revenue Per Share}
\[
    \text{Revenue\_Per\_Share} = \frac{\text{Revenue\_LTM}}{\text{Shares\_Outstanding}}
\]

\subsection{NNI Margin}
\[
    \text{NNI\_Margin} = \frac{\text{NNI\_TTM}}{\text{Revenue\_Per\_Share}} = \frac{\text{NI\_TTM}}{\text{Revenue\_LTM}}
\]
where $\text{NNI\_Margin}$ is the Normalized Net Income Margin (Net Profit Margin).

\paragraph{Signal:}
\[
    \text{NNI\_Margin} > 0
\]
